const express = require("express"); const app = express(); const cors = require("cors"); app.use(cors()); app.use(express.json()); /* ========================= 📦 데이터 저장 (메모리) ========================= */ let pets = []; /* ========================= 🌐 API ========================= */ /* 목록 */ app.get("/api/pets", (req, res) => { res.json(pets); }); /* 등록 */ app.post("/api/pets", (req, res) => { const pet = { id: Date.now(), name: req.body.name, type: req.body.type, sex: req.body.sex, info: req.body.info, like: false }; pets.push(pet); res.json(pet); }); /* 수정 */ app.put("/api/pets/:id", (req, res) => { const id = Number(req.params.id); pets = pets.map(p => p.id === id ? { ...p, ...req.body } : p ); res.json({ ok: true }); }); /* 삭제 */ app.delete("/api/pets/:id", (req, res) => { const id = Number(req.params.id); pets = pets.filter(p => p.id !== id); res.json({ ok: true }); }); /* ========================= 🌐 프론트 HTML 서빙 ========================= */ app.get("/", (req, res) => { res.send(` 댕냥이 분양센터
🐶🐱 댕냥이 분양센터 (통합버전)
`); }); /* ========================= 🚀 서버 실행 ========================= */ app.listen(3000, () => { console.log("서버 실행: http://localhost:3000"); });
